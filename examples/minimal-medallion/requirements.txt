# Local dev dependencies for the minimal-medallion example.
#
# Versions are pinned to Microsoft Fabric Runtime 1.3 (Spark 3.5 / Delta 3.2)
# so local execution matches what runs in Fabric. pyfabric-dev itself only
# floors these (pyspark>=3.5, delta-spark>=3.2); without the pins below, pip
# would resolve to the latest (Spark 4.x / Delta 4.x), which no Fabric runtime
# uses — so always install via this file, not a bare `pip install pyfabric-dev`.

pyfabric-dev>=0.4.1

# Spark + Delta — match Fabric Runtime 1.3.
pyspark==3.5.7
delta-spark==3.2.1

# Data processing — major versions match Fabric Runtime 1.3 (pandas 2.x, numpy 1.x).
# Exact Fabric pins (pandas 2.1.4, numpy 1.26.4) require Python 3.11; local dev uses
# 3.13, so pandas is floored at 2.3 for pre-built cp313 wheels.
pandas>=2.3.3,<3.0
numpy>=1.26.4,<2.0

# Testing
pytest>=7.0.0
